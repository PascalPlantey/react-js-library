Enum

new Enum(name, values) → {object}

Create a pseudo enumeration.

Description:

  • Create a pseudo enumeration. The following values cannot be used as key name: ['get', 'has', 'keys', 'name', 'protected', 'toString' , 'values']

Source:
  • js/extensions/Enum.js, line 25
Examples
const directions = createEnum('Directions', ['up', 'down']);
console.log(directions.up);                  // 'up'
for(const [key, value] of directions)
  console.log(key, value);                   // 'up', 'up' / 'down', 'down' }
const keyboardDirections = createEnum('Keyboard directions', [
  { key: 'up', value: 56 },
  ['down', 84],
  ['left', 55],
  ['right', 57]
]);
switch(key) {
  case(keyboardDirections.up)
    // code
    break;
  ...
}
